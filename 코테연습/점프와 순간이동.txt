using System;

class Solution
{
    public int solution(int n)
    {
        // 슈트개발 : k칸 앞점프, 현재거리 X 2위치로 순간이동
        // 건전지로 작동 : 순간이동은 건전지 안줄어듦, 점프는 줄어듦
        // k칸 점프하면 k만큼 깎임
        // N까지 가려고함, 점프는 최소한으로, 건전지 사용량의 최소값을 구해라
        // N까지 k점프와 순간이동을 사용해서 최소한의 K로 도착해라

        // n을 2로 나눈 나머지가 0이면 순간이동, 아니면 점프
        // => n을 2로 나눈 나머지 만큼 점프
        int answer = 0;
        while(n > 0)
        {
            answer += n % 2;
            n /= 2;
        }
        return answer;
    }
}
