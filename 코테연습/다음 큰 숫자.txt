using System;

class Solution 
{
    public int solution(int n) 
   {
        int answer = n + 1;
        while(count(n) != count(answer))
            answer++;
        return answer;
    }
    public int count(int n)
    {
        int cnt = 0;
        while(n != 0)
        {
            if (n % 2 == 1) cnt++;
            n = n / 2;
        }
        return cnt;
    }
}
